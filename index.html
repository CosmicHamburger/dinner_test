    <div class="container">

        <h2>🍚 今晚吃啥？抽奖啦</h2>

        <div class="input-box">

            <input type="text" id="dinnerInput" placeholder="输入晚餐名称（如：火锅、烧烤）">

            <button class="btn-add" onclick="addDinner()">添加</button>

        </div>

        <div class="list" id="dinnerList"></div>

        <div class="result" id="result"></div>

        <div class="input-box">

            <button class="btn-draw" onclick="luckyDraw()">开始抽奖</button>

            <button class="btn-clear" onclick="clearAll()">清空全部</button>

        </div>

    </div>


    <script>

        // 本地存储：刷新不丢失数据

        let dinnerList = JSON.parse(localStorage.getItem('dinnerList')) || [];

        const input = document.getElementById('dinnerInput');

        const listDom = document.getElementById('dinnerList');

        const resultDom = document.getElementById('result');


        // 渲染列表

        function renderList() {

            listDom.innerHTML = dinnerList.map(item => `<div class="item">${item}</div>`).join('');

            resultDom.innerText = '';

            // 保存到本地

            localStorage.setItem('dinnerList', JSON.stringify(dinnerList));

        }


        // 添加晚餐

        function addDinner() {

            const value = input.value.trim();

            if (!value) return alert('请输入晚餐名称！');

            if (dinnerList.includes(value)) return alert('已经添加过啦！');

            dinnerList.push(value);

            renderList();

            input.value = '';

        }


        // 随机抽奖

        function luckyDraw() {

            if (dinnerList.length < 2) return alert('至少添加2个选项才能抽奖哦！');

            const randomIndex = Math.floor(Math.random() * dinnerList.length);

            const winner = dinnerList[randomIndex];

            resultDom.innerText = `🎉 中奖：${winner}！`;

            

            // 高亮效果

            const items = document.querySelectorAll('.item');

            items.forEach((el, idx) => {

                el.classList.toggle('win', idx === randomIndex);

            });

        }


        // 清空

        function clearAll() {

            dinnerList = [];

            renderList();

        }


        // 初始化页面

        renderList();

    </script>
